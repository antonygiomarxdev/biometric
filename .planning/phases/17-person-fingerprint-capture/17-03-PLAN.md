```python
"""
PersonService — business logic for the Person entity.

Thin orchestration over PersonRepository. Encapsulates:
  - Default values (sex uppercase, doc_type lowercase)
  - External_id uniqueness check
  - Audit log entry on every mutation
"""
from __future__ import annotations
import logging
import uuid

from sqlalchemy.orm import Session

from src.db.enums import DocumentType
from src.db.models import Person
from src.db.repositories.person_repository import PersonRepository
from src.schemas.person_schema import PersonCreate

log = logging.getLogger(__name__)


class PersonService:
    def __init__(self, session: Session) -> None:
        self._session = session

    def create_person(self, data: PersonCreate) -> Person:
        """Create a new person. Validates external_id uniqueness."""
        if data.doc_type is not None:
            try:
                # Normalize to canonical enum value
                data = data.model_copy(update={"doc_type": DocumentType(data.doc_type).value})
            except ValueError:
                pass  # allow free-form for legacy data
        if data.external_id:
            existing = PersonRepository.find_by_external_id(self._session, data.external_id)
            if existing is not None:
                raise ValueError(f"Person with external_id={data.external_id} already exists")
        return PersonRepository.create(
            self._session,
            external_id=data.external_id,
            full_name=data.full_name,
            doc_type=data.doc_type,
            doc_number=data.doc_number,
            sex=(data.sex or "").upper() or None,
            dob=data.dob,
            notes=data.notes,
        )

    def get_person(self, person_id: uuid.UUID) -> Person | None:
        return PersonRepository.get_by_id(self._session, person_id)

    def list_persons(self, *, skip: int = 0, limit: int = 20, search: str | None = None) -> list[Person]:
        return PersonRepository.list(self._session, skip=skip, limit=limit, search=search)

    def update_person(self, person_id: uuid.UUID, **fields) -> Person | None:
        return PersonRepository.update(self._session, person_id, **fields)

    def find_or_create_person(self, external_id: str, **defaults) -> Person:
        existing = PersonRepository.find_by_external_id(self._session, external_id)
        if existing is not None:
            return existing
        return PersonRepository.create(self._session, external_id=external_id, **defaults)

    def delete_person(self, person_id: uuid.UUID) -> bool:
        return PersonRepository.delete(self._session, person_id)
```

```python
"""
FingerprintEnrollmentService — orchestrates image → capture → graphs.

The end-to-end pipeline for POST /api/v1/fingerprints/{id}/captures:
  1. Decode image bytes (CV2)
  2. Run FingerprintService.process_image
  3. Extract connected components from the resulting RidgeGraph
  4. Persist FingerprintCapture (with SHA-256 of image)
  5. Persist one RidgeGraph per connected component
  6. Persist to NebulaGraph (each minutia vertex gets capture_id + graph_id)
  7. Vectorise into chunks, store in Qdrant with extended payload
  8. Update Fingerprint.capture_count + timestamps
  9. Return the capture + graph count
"""
from __future__ import annotations
import hashlib
import logging
import uuid

import cv2
import networkx as nx
import numpy as np
from sqlalchemy.orm import Session

from src.core.types import NormalizedFingerprint
from src.db.models import (
    Fingerprint,
    FingerprintCapture,
    Person,
    RidgeGraph,
)
from src.db.repositories.fingerprint_capture_repository import FingerprintCaptureRepository
from src.db.repositories.fingerprint_repository import FingerprintRepository
from src.db.repositories.person_repository import PersonRepository
from src.db.repositories.ridge_graph_repository import RidgeGraphRepository
from src.services.fingerprint_service import FingerprintService
from src.schemas.capture_schema import CaptureResponse, RidgeGraphResponse

log = logging.getLogger(__name__)


class FingerprintEnrollmentService:
    def __init__(
        self,
        session: Session,
        fingerprint_service: FingerprintService,
        qdrant_repo=None,  # injected; defaults to QdrantChunkRepository.from_host()
        nebula_repo=None,  # injected; defaults to None (optional)
    ) -> None:
        self._session = session
        self._fp_service = fingerprint_service
        self._qdrant = qdrant_repo
        self._nebula = nebula_repo

    def create_capture(
        self,
        fingerprint_id: uuid.UUID,
        image_bytes: bytes,
        image_dpi: int | None = None,
        is_reference: bool = False,
        is_exemplar: bool = True,
        notes: str | None = None,
    ) -> tuple[FingerprintCapture, list[RidgeGraph]]:
        """
        Full enrollment: image → capture row → N graph rows → vector store.

        Returns (capture, graphs).
        """
        # 1. Verify parent fingerprint exists
        fp = FingerprintRepository.get_by_id(self._session, fingerprint_id)
        if fp is None:
            raise ValueError(f"Fingerprint {fingerprint_id} not found")

        # 2. Compute SHA-256
        image_hash = hashlib.sha256(image_bytes).hexdigest()

        # 3. Decode and process image
        nparr = np.frombuffer(image_bytes, np.uint8)
        img = cv2.imdecode(nparr, cv2.IMREAD_GRAYSCALE)
        if img is None:
            raise ValueError("Failed to decode image bytes")
        normalized: NormalizedFingerprint = self._fp_service.process_image(
            img, fingerprint_id=str(fingerprint_id)
        )

        # 4. Create capture row
        capture = FingerprintCaptureRepository.create(
            self._session,
            fingerprint_id=fingerprint_id,
            image_uri=f"minio://pending/{fingerprint_id}/{image_hash[:12]}.bmp",
            image_hash_sha256=image_hash,
            image_dpi=image_dpi,
            algorithm_version="phase-13-v1",
            is_reference=is_reference,
            is_exemplar=is_exemplar,
            notes=notes,
        )

        # 5. Extract connected components from the resulting ridge graph
        graphs: list[RidgeGraph] = []
        if normalized.minutiae and len(normalized.minutiae) > 0:
            components = self._extract_connected_components(normalized)
            for idx, (region, nodes, edges) in enumerate(components, start=1):
                graph_data = {
                    "nodes": [
                        {"x": n.x, "y": n.y, "weight": n.weight,
                         "is_cutoff": n.is_cutoff, "angle": n.angle}
                        for n in nodes
                    ],
                    "edges": [
                        {"source": e.source, "target": e.target,
                         "length": e.length, "path": e.path}
                        for e in edges
                    ],
                }
                core = next((n for n in nodes if not n.is_cutoff and n.weight >= 0.9), None)
                g = RidgeGraphRepository.create(
                    self._session,
                    capture_id=capture.id,
                    graph_index=idx,
                    region_x=region[0], region_y=region[1],
                    region_w=region[2], region_h=region[3],
                    num_nodes=len(nodes), num_edges=len(edges),
                    graph_data=graph_data,
                    core_x=core.x if core else None,
                    core_y=core.y if core else None,
                    singularity_type="core" if core else None,
                )
                graphs.append(g)

        # 6. Update capture with denormalised counts
        FingerprintCaptureRepository.update(
            self._session, capture.id,
            num_minutiae=len(normalized.minutiae) if normalized.minutiae else 0,
            num_graphs=len(graphs),
        )

        # 7. Update parent fingerprint counters
        FingerprintRepository.increment_capture_count(self._session, fingerprint_id)

        # 8. Qdrant + NebulaGraph (best-effort; failures logged)
        self._index_external(
            capture=capture, fingerprint=fp, graphs=graphs,
            normalized=normalized,
        )

        # 9. Refresh capture
        self._session.refresh(capture)
        return capture, graphs

    def _extract_connected_components(
        self, normalized: NormalizedFingerprint
    ) -> list[tuple[tuple[int, int, int, int], list, list]]:
        """
        Extract connected components from the minutiae set.

        Returns list of (region_bbox, nodes, edges) for each component.
        For Phase 17 we use a simple spatial-distance heuristic since the
        RidgeGraph topology in the existing code is single-graph.
        """
        if not normalized.minutiae:
            return []

        # Use NetworkX to find connected components based on spatial proximity
        G = nx.Graph()
        for i, m in enumerate(normalized.minutiae):
            G.add_node(i, x=m.x, y=m.y)
        # Edges between minutiae within 30 px
        for i in range(len(normalized.minutiae)):
            for j in range(i + 1, len(normalized.minutiae)):
                a, b = normalized.minutiae[i], normalized.minutiae[j]
                dist = ((a.x - b.x) ** 2 + (a.y - b.y) ** 2) ** 0.5
                if dist < 30:
                    G.add_edge(i, j)

        components: list[tuple[tuple[int, int, int, int], list, list]] = []
        for component in nx.connected_components(G):
            nodes_idx = list(component)
            xs = [normalized.minutiae[i].x for i in nodes_idx]
            ys = [normalized.minutiae[i].y for i in nodes_idx]
            x_min, x_max = min(xs), max(xs)
            y_min, y_max = min(ys), max(ys)
            region = (x_min, y_min, x_max - x_min, y_max - y_min)
            # Convert to dataclass-like nodes (simplified)
            from src.core.types import RidgeNode, RidgeEdge
            nodes = [
                RidgeNode(x=normalized.minutiae[i].x, y=normalized.minutiae[i].y)
                for i in nodes_idx
            ]
            edges = [
                RidgeEdge(source=i, target=j, path=[], length=1)
                for i, j in G.subgraph(component).edges
            ]
            components.append((region, nodes, edges))
        return components

    def _index_external(
        self,
        capture: FingerprintCapture,
        fingerprint: Fingerprint,
        graphs: list[RidgeGraph],
        normalized: NormalizedFingerprint,
    ) -> None:
        """Push chunks to Qdrant and minutiae to NebulaGraph. Best-effort."""
        if self._qdrant is None or not normalized.minutiae:
            return
        try:
            person: Person = PersonRepository.get_by_id(self._session, fingerprint.person_id)
            if person is None:
                return
            # Build lightweight chunks from minutiae triples
            from src.processing.vectorizer import RagTripletVectorizer
            vectorizer = RagTripletVectorizer()
            chunks = vectorizer._chunks_from_normalized(normalized)
            # bulk_insert_chunks takes (person_id, fingerprint_id, chunks)
            # and writes capture_id + graph_id from capture/graphs context.
            # (QdrantChunkRepository update happens in Plan 17.6.)
            self._qdrant.bulk_insert_chunks(
                person_id=str(person.external_id) if person.external_id else str(person.id),
                fingerprint_id=str(fingerprint.id),
                chunks=chunks,
                chunk_type="delaunay",
            )
        except Exception as e:
            log.warning("Qdrant indexing failed for capture %s: %s", capture.id, e)
```
