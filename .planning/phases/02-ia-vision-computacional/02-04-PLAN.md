    """
    Deep Learning minutiae extraction inference.
    
    Handles pre/post processing for a neural minutiae detection model.
    The model output is decoded into MinutiaCandidate instances.
    """
    import logging
    import numpy as np
    from src.core.types import MinutiaCandidate, MinutiaType, AlgorithmOrigin, Confidence
    from src.ai.config import AiConfig
    from src.ai.model_manager import ModelManager

    logger = logging.getLogger(__name__)

    class ExtractionProcessor:
        """Pre/post processing for DL minutiae extraction model."""

        def __init__(self, config: AiConfig) -> None:
            self.config = config

        def preprocess(self, image: np.ndarray) -> np.ndarray:
            """Normalize and prepare image for model inference.
            
            Args:
                image: Grayscale uint8 image (H, W) or (H, W, 1).
            Returns:
                Float32 tensor (1, 1, 512, 512) normalized to [0, 1].
            """
            if image.ndim == 3:
                if image.shape[2] == 1:
                    image = image[:, :, 0]
                elif image.shape[2] == 3:
                    import cv2
                    image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
            # Normalize to [0, 1]
            img_float = image.astype(np.float32) / 255.0
            h, w = img_float.shape
            size = self.config.input_size
            # Center-pad to square
            padded = np.zeros((size, size), dtype=np.float32)
            y_offset = (size - min(h, size)) // 2
            x_offset = (size - min(w, size)) // 2
            padded[y_offset:y_offset + min(h, size), x_offset:x_offset + min(w, size)] = \
                img_float[:min(h, size), :min(w, size)]
            return padded[np.newaxis, np.newaxis, ...]  # (1, 1, 512, 512)

        def postprocess(
            self, raw_output: np.ndarray, original_shape: tuple[int, int],
            confidence_threshold: float | None = None,
        ) -> list[MinutiaCandidate]:
            """Decode model output into MinutiaCandidate list.
            
            Expected model output format:
            - Heatmap layers for terminations and bifurcations (2, H, W)
            - Each pixel's value = confidence for that minutia type
            
            Args:
                raw_output: Model output tensor, shape depends on model architecture.
                    Expected: (1, 2, 512, 512) or (2, 512, 512) with channels [terminations, bifurcations].
                original_shape: (H, W) of the original input image.
                confidence_threshold: Minimum confidence to accept a detection.
            
            Returns:
                List of MinutiaCandidate with AlgorithmOrigin.DEEP_LEARNING.
            """
            threshold = confidence_threshold or self.config.confidence_threshold
            # Squeeze batch dim
            output = raw_output.squeeze()  # (2, 512, 512) or (512, 512)
            if output.ndim == 2:
                # Single-channel output, assume general minutiae
                heatmap = output
                candidates = self._decode_heatmap(heatmap, threshold, MinutiaType.UNKNOWN)
            elif output.ndim == 3 and output.shape[0] >= 2:
                # Multi-channel: [terminations, bifurcations, ...]
                term_heatmap = output[0]
                bif_heatmap = output[1]
                terminations = self._decode_heatmap(term_heatmap, threshold, MinutiaType.TERMINATION)
                bifurcations = self._decode_heatmap(bif_heatmap, threshold, MinutiaType.BIFURCATION)
                candidates = terminations + bifurcations
            else:
                logger.warning("Unexpected model output shape: %s", output.shape)
                return []

            # Adjust coordinates for padding offset
            h_orig, w_orig = original_shape
            size = self.config.input_size
            y_offset = (size - min(h_orig, size)) // 2
            x_offset = (size - min(w_orig, size)) // 2
            scale_y = min(h_orig, size) / size
            scale_x = min(w_orig, size) / size

            adjusted: list[MinutiaCandidate] = []
            for m in candidates:
                # Reverse the padding offset
                raw_x = m.x / scale_x if scale_x > 0 else m.x
                raw_y = m.y / scale_y if scale_y > 0 else m.y
                x_adj = int(raw_x - x_offset)
                y_adj = int(raw_y - y_offset)
                if 0 <= x_adj < w_orig and 0 <= y_adj < h_orig:
                    adjusted.append(MinutiaCandidate(
                        x=x_adj, y=y_adj,
                        angle=m.angle, type=m.type,
                        confidence=m.confidence,
                        origin=AlgorithmOrigin.DEEP_LEARNING,
                    ))

            logger.info("DL extraction: %d candidates (terminations: %d, bifurcations: %d)",
                        len(adjusted),
                        sum(1 for m in adjusted if m.type == MinutiaType.TERMINATION),
                        sum(1 for m in adjusted if m.type == MinutiaType.BIFURCATION))
            return adjusted

        def _decode_heatmap(
            self, heatmap: np.ndarray,
            threshold: float,
            mtype: MinutiaType,
        ) -> list[MinutiaCandidate]:
            """Extract local maxima from a heatmap above confidence threshold."""
            from scipy.ndimage import maximum_filter, label

            # Non-maximum suppression: only keep local maxima
            neighborhood = 3
            local_max = (heatmap == maximum_filter(heatmap, size=neighborhood))
            detected = (heatmap > threshold) & local_max

            if not np.any(detected):
                return []

            # Label connected components
            labeled, num_features = label(detected)
            candidates: list[MinutiaCandidate] = []
            for feat_id in range(1, num_features + 1):
                ys, xs = np.where(labeled == feat_id)
                if len(xs) == 0:
                    continue
                # Center of mass for sub-pixel accuracy
                region_vals = heatmap[ys, xs]
                total = region_vals.sum()
                if total == 0:
                    continue
                cx = float(np.average(xs, weights=region_vals))
                cy = float(np.average(ys, weights=region_vals))
                confidence = float(region_vals.max())
                candidates.append(MinutiaCandidate(
                    x=int(round(cx)),
                    y=int(round(cy)),
                    angle=0.0,  # Angle can be computed post-hoc from local ridge orientation
                    type=mtype,
                    confidence=confidence,
                    origin=AlgorithmOrigin.DEEP_LEARNING,
                ))
            return candidates
    ```

    class AiFeatureExtractor(IFeatureExtractor):
        """Deep Learning minutiae extractor using ONNX Runtime."""

        def __init__(
            self,
            model_manager: ModelManager,
            processor: ExtractionProcessor | None = None,
        ) -> None:
            self.model_manager = model_manager
            self.processor = processor or ExtractionProcessor(AiConfig())

        @timed("extract_minutiae_dl")
        def extract(self, image: np.ndarray) -> list[MinutiaCandidate]:
            """Extract minutiae using DL model inference.
            
            Args:
                image: Grayscale uint8 image.
            Returns:
                List of MinutiaCandidates with AlgorithmOrigin.DEEP_LEARNING.
            """
            import cv2
            if image is None or image.size == 0:
                logger.warning("Empty image passed to AiFeatureExtractor")
                return []
            if image.ndim == 3 and image.shape[2] == 3:
                image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
            if image.ndim == 3 and image.shape[2] == 1:
                image = image[:, :, 0]

            original_shape = image.shape[:2]
            try:
                input_tensor = self.processor.preprocess(image)
                # Non-blocking GPU inference via ModelManager
                raw_output = self.model_manager.run_extraction(input_tensor)
                candidates = self.processor.postprocess(raw_output, original_shape)
                return candidates
            except Exception as e:
                logger.error("DL extraction failed: %s", e, exc_info=True)
                return []
    ```
