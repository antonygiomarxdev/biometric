  <name>Task 3: Implement ModelManager Singleton</name>
  <files>apps/backend/src/ai/model_manager.py</files>
  <action>
    Create `apps/backend/src/ai/model_manager.py` with the `ModelManager` class per RESEARCH.md Model Manager Pattern:

    ```python
    import logging
    from pathlib import Path
    import numpy as np
    import onnxruntime as ort
    from src.ai.config import AiConfig

    logger = logging.getLogger(__name__)

    class ModelManager:
        """Manages ONNX Runtime inference sessions for all AI models.
        
        Singleton loaded in FastAPI lifespan. Provides session caching,
        GPU/CPU provider selection, and typed inference methods.
        """

        def __init__(self, config: AiConfig) -> None:
            self.config = config
            self._sessions: dict[str, ort.InferenceSession] = {}
            self.model_dir = Path(config.model_dir)
            self.model_dir.mkdir(parents=True, exist_ok=True)

        def load_model(self, name: str) -> ort.InferenceSession:
            """Load an ONNX model from disk, cache the session."""
            if name in self._sessions:
                return self._sessions[name]
            path = self.model_dir / f"{name}.onnx"
            if not path.exists():
                raise FileNotFoundError(f"ONNX model not found: {path}")
            session = ort.InferenceSession(
                str(path),
                providers=[self.config.provider, "CPUExecutionProvider"],
            )
            self._sessions[name] = session
            logger.info("Loaded ONNX model '%s' (%s, %d inputs)", name, self.config.provider, len(session.get_inputs()))
            return session

        def get_session(self, name: str) -> ort.InferenceSession:
            """Get a cached session, loading if necessary."""
            return self.load_model(name)

        def unload_model(self, name: str) -> None:
            """Remove a session from cache."""
            self._sessions.pop(name, None)

        def unload_all(self) -> None:
            """Clear all cached sessions."""
            self._sessions.clear()

        @property
        def loaded_models(self) -> list[str]:
            return list(self._sessions.keys())

        # --- Typed inference helpers ---

        def _run_single(self, name: str, input_tensor: np.ndarray) -> np.ndarray:
            """Run inference on a single input tensor."""
            session = self.get_session(name)
            input_name = session.get_inputs()[0].name
            output_name = session.get_outputs()[0].name
            result = session.run([output_name], {input_name: input_tensor})
            return result[0]

        def run_segmentation(self, image: np.ndarray) -> np.ndarray:
            """Run segmentation model, return binary mask."""
            return self._run_single("segment", image)

        def run_enhancement(self, image: np.ndarray) -> np.ndarray:
            """Run enhancement model, return enhanced image."""
            return self._run_single("enhance", image)

        def run_extraction(self, image: np.ndarray) -> np.ndarray:
            """Run extraction model, return raw model output."""
            return self._run_single("extract", image)
    ```
